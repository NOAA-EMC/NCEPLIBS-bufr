4 C> THIS SUBROUTINE PARSES THE TABLE A MNEMONIC AND THE DATE
5 C> OUT OF SECTION 1 OF A BUFR MESSAGE PREVIOUSLY READ FROM UNIT LUNIT
6 C> USING BUFR ARCHIVE LIBRARY SUBROUTINE READMG OR EQUIVALENT (AND NOW
7 C> STORED IN THE INTERNAL MESSAGE BUFFER, ARRAY MBAY IN MODULE
8 C> BITBUF). THE TABLE A MNEMONIC IS ASSOCIATED WITH THE BUFR
9 C> MESSAGE TYPE/SUBTYPE IN SECTION 1. IT ALSO FILLS IN THE MESSAGE
10 C> CONTROL WORD PARTITION ARRAYS IN MODULE MSGCWD.
11 C>
12 C> PROGRAM HISTORY LOG:
13 C> 2000-09-19 J. WOOLLEN -- ORIGINAL AUTHOR - CONSOLIDATED MESSAGE
14 C> DECODING LOGIC THAT HAD BEEN REPLICATED IN
15 C> READMG, READFT, READERME, RDMEMM AND READIBM
16 C> (CKTABA IS NOW CALLED BY THESE CODES);
17 C> LOGIC ENHANCED HERE TO ALLOW COMPRESSED AND
18 C> STANDARD BUFR MESSAGES TO BE READ
19 C> 2003-11-04 S. BENDER -- ADDED REMARKS/BUFRLIB ROUTINE
20 C> INTERDEPENDENCIES
21 C> 2003-11-04 D. KEYSER -- MODIFIED TO NOT ABORT WHEN THE SECTION 1
22 C> MESSAGE SUBTYPE DOES NOT AGREE WITH THE
23 C> SECTION 1 MESSAGE SUBTYPE IN THE DICTIONARY
24 C> IF THE MESSAGE TYPE MNEMONIC IS NOT OF THE
25 C> FORM "NCtttsss", WHERE ttt IS THE BUFR TYPE
26 C> AND sss IS THE BUFR SUBTYPE (E.G., IN
27 C> "PREPBUFR" FILES); MODIFIED DATE
28 C> CALCULATIONS TO NO LONGER USE FLOATING
29 C> POINT ARITHMETIC SINCE THIS CAN LEAD TO
30 C> ROUND OFF ERROR AND AN IMPROPER RESULTING

75  SUBROUTINE cktaba(LUN,SUBSET,JDATE,IRET)

77 
78 
79  USE moda_msgcwd
80  USE moda_sc3bfr
81  USE moda_unptyp
82  USE moda_bitbuf
83 
84  include 'bufrlib.prm'
85 
86  COMMON /padesc/ ibct,ipd1,ipd2,ipd3,ipd4
87  COMMON /quiet / iprt
88 
89  CHARACTER*128 BORT_STR,ERRSTR
90  CHARACTER*8 SUBSET
91  CHARACTER*2 CPFX(3)
92  CHARACTER*1 TAB
93  LOGICAL TRYBT, DIGIT
94 
95  DATA cpfx / 'NC', 'FR', 'FN' /
96  DATA ncpfx / 3 /
97 
98 C-----------------------------------------------------------------------
99 C-----------------------------------------------------------------------
100 
101  iret = 0
102 
103  trybt = .true.
104 
105  jdate = igetdate(mbay(1,lun),iyr,imo,idy,ihr)
106 
107 c .... Message type
108  mtyp = iupbs01(mbay(1,lun),'MTYP')
109 c .... Message subtype
110  msbt = iupbs01(mbay(1,lun),'MSBT')
111 
112  IF(mtyp.EQ.11) THEN
113 c .... This is a BUFR table (dictionary) message.
114  iret = 11
115 c .... There's no need to proceed any further unless Section 3 is being
116 c .... used for decoding.
117  IF(isc3(lun).EQ.0) THEN
118  subset = " "
119  GOTO 100
120  ENDIF
121  ENDIF
122 
123 C PARSE SECTION 3
124 C ---------------
125 
126  CALL getlens(mbay(1,lun),3,len0,len1,len2,len3,l4,l5)
127 
128  iad3 = len0+len1+len2
129 
130 c .... First descriptor (integer)
131  ksub = iupb(mbay(1,lun),iad3+8 ,16)
132 c .... Second descriptor (integer)
133  isub = iupb(mbay(1,lun),iad3+10,16)
134 
135 C LOCATE SECTION 4
136 C ----------------
137 
138  iad4 = iad3+len3
139 
140 C NOW, TRY TO GET "SUBSET" (MNEMONIC ASSOCIATED WITH TABLE A) FROM MSG
141 C --------------------------------------------------------------------
142 
143 C FIRST CHECK WHETHER SECTION 3 IS BEING USED FOR DECODING
144 C --------------------------------------------------------
145 
146  IF(isc3(lun).NE.0) THEN
147  subset = tamnem(lun)
148 c .... is SUBSET from Table A?
149  CALL nemtbax(lun,subset,mty1,msb1,inod)
150  IF(inod.GT.0) THEN
151 c .... yes it is
152  mbyt(lun) = 8*(iad4+4)
153  msgunp(lun) = 1
154  GOTO 10
155  ENDIF
156  ENDIF
157 
158 C IF ISUB FROM SECTION 3 DEFINES TABLE A THEN MSGUNP=0
159 C ----------------------------------------------------
160 
161 c .... get SUBSET from ISUB
162 5 CALL numtab(lun,isub,subset,tab,itab)
163 c .... is SUBSET from Table A?
164  CALL nemtbax(lun,subset,mty1,msb1,inod)
165  IF(inod.GT.0) THEN
166 c .... yes it is
167  mbyt(lun) = (iad4+4)
168  msgunp(lun) = 0
169  GOTO 10
170  ENDIF
171 
172 C IF KSUB FROM SECTION 3 DEFINES TABLE A THEN MSGUNP=1 (standard)
173 C ---------------------------------------------------------------
174 
175 c .... get SUBSET from KSUB
176  CALL numtab(lun,ksub,subset,tab,itab)
177 c .... is SUBSET from Table A?
178  CALL nemtbax(lun,subset,mty1,msb1,inod)
179  IF(inod.GT.0) THEN
180 c .... yes it is
181  mbyt(lun) = 8*(iad4+4)
182  msgunp(lun) = 1
183  GOTO 10
184  ENDIF
185 
186 C OKAY, STILL NO "SUBSET", LETS MAKE IT "NCtttsss" (where ttt=MTYP
187 C and sss=MSBT) AND SEE IF IT DEFINES TABLE A. IF NOT, THEN ALSO
188 C TRY "FRtttsss" AND "FNtttsss".
189 C ----------------------------------------------------------------
190 
191  ii=1
192  DO WHILE(ii.LE.ncpfx)
193  WRITE(subset,'(A2,2I3.3)') cpfx(ii),mtyp,msbt
194 c .... is SUBSET from Table A?
195  CALL nemtbax(lun,subset,mty1,msb1,inod)
196  IF(inod.GT.0) THEN
197 c .... yes it is
198  IF(ksub.EQ.ibct) THEN
199  mbyt(lun) = (iad4+4)
200  msgunp(lun) = 0
201  ELSE
202  mbyt(lun) = 8*(iad4+4)
203  msgunp(lun) = 1
204  ENDIF
205  GOTO 10
206  ENDIF
207  ii=ii+1
208  ENDDO
209 
210 C NOW WE HAVE A GENERATED "SUBSET", BUT IT STILL DOES NOT DEFINE
211 C TABLE A - MAKE ONE LAST DESPERATE ATTEMPT - SEE IF AN EXTERNAL
212 C USER-SUPPLIED BUFR DICTIONARY TABLE IN CHARACTER FORMAT IS DEFINED
213 C IN OPENBT (ONLY POSSIBLE IF APPLICATION PROGRAM HAS AN IN-LINE
214 C OPENBT OVERRIDING THE ONE IN THE BUFR ARCHIVE LIBRARY)
215 C ------------------------------------------------------------------
216 
217  IF(trybt) THEN
218  trybt = .false.
219  IF(iprt.GE.1) THEN
220  CALL errwrt('++++++++++++++BUFR ARCHIVE LIBRARY+++++++++++++++++')
221  errstr = 'BUFRLIB: CKTABA - LAST RESORT, CHECK FOR EXTERNAL'//
222  . ' BUFR TABLE VIA CALL TO IN-LINE OPENBT'
223  CALL errwrt(errstr)
224  CALL errwrt('++++++++++++++BUFR ARCHIVE LIBRARY+++++++++++++++++')
225  CALL errwrt(' ')
226  ENDIF
227  CALL openbt(lundx,mtyp)
228  IF(lundx.GT.0) THEN
229 c .... Good news, there is a unit (LUNDX) connected to a table file,
230 c .... so store the table internally
231  CALL rdusdx(lundx,lun)
232  GOTO 5
233  ENDIF
234  ENDIF
235 
236 C IF ALL ATTEMPTS TO DEFINE TABLE A FAIL SKIP GIVE UP
237 C ---------------------------------------------------
238 
239  IF(iprt.GE.0) THEN
240  CALL errwrt('+++++++++++++++++++++WARNING+++++++++++++++++++++++')
241  errstr = 'BUFRLIB: CKTABA - UNRECOGNIZED TABLE A MESSAGE TYPE ('//
242  . subset // ') - RETURN WITH IRET = -1'
243  CALL errwrt(errstr)
244  CALL errwrt('+++++++++++++++++++++WARNING+++++++++++++++++++++++')
245  CALL errwrt(' ')
246  ENDIF
247  iret = -1
248  GOTO 100
249 
250 C CHECK THE VALIDITY OF THE MTYP/MSBT AND FOR COMPRESSION (MSGUNP=2)
251 C ------------------------------------------------------------------
252 
253 10 IF(isc3(lun).EQ.0) THEN
254  IF(mtyp.NE.mty1) GOTO 900
255  IF(msbt.NE.msb1.AND.digit(subset(3:8))) GOTO 901
256  ENDIF
257  IF(iupbs3(mbay(1,lun),'ICMP').GT.0) msgunp(lun) = 2
258 
259 C SET THE OTHER REQUIRED PARAMETERS IN MESSAGE CONTROL WORD PARTITION
260 C -------------------------------------------------------------------
261 
262 c .... Date for this message
263  idate(lun) = i4dy(jdate)
264 c .... Positional index of Table A mnem.
265  inode(lun) = inod
266 c .... Number of subsets in this message
267  msub(lun) = iupbs3(mbay(1,lun),'NSUB')
268 c .... Number of subsets read so far from this message
269  nsub(lun) = 0
270 
271  IF(iret.NE.11) THEN
272 c .... Number of non-dictionary messages read so far from this file
273  nmsg(lun) = nmsg(lun)+1
274  ENDIF
275 
276 C EXITS
277 C -----
278 
279 100 RETURN
280 900 WRITE(bort_str,'("BUFRLIB: CKTABA - MESSAGE TYPE MISMATCH '//
281  . '(SUBSET=",A8,", MTYP=",I3,", MTY1=",I3)') subset,mtyp,mty1
282  CALL bort(bort_str)
283 901 WRITE(bort_str,'("BUFRLIB: CKTABA - MESSAGE SUBTYPE MISMATCH '//
284  . '(SUBSET=",A8,", MSBT=",I3,", MSB1=",I3)') subset,msbt,msb1
285  CALL bort(bort_str)
286  END
