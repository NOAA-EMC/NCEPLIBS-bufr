5 C> BUFR EDITION 3 AND OUTPUTS AN EQUIVALENT BUFR MESSAGE ENCODED USING

14 C>

16 C> INPUT ARGUMENT LIST:
17 C> MSGIN - INTEGER: *-WORD ARRAY CONTAINING BUFR MESSAGE ENCODED
18 C> USING BUFR EDITION 3

68 C Check for overflow of the output array. Note that the new

70 C input message (i.e. 4 more bytes in Section 1, but 1 fewer
71 C byte in Section 3).

84  CALL pkb ( 4, 8, msgot, ibit )

99  CALL pkb ( iupbs01(msgin,'MSBT'), 8, msgot, ibit )
100  CALL pkb ( iupbs01(msgin,'MTV'), 8, msgot, ibit )
101  CALL pkb ( iupbs01(msgin,'MTVL'), 8, msgot, ibit )
102  CALL pkb ( iupbs01(msgin,'YEAR'), 16, msgot, ibit )
103  CALL pkb ( iupbs01(msgin,'MNTH'), 8, msgot, ibit )
104  CALL pkb ( iupbs01(msgin,'DAYS'), 8, msgot, ibit )
105  CALL pkb ( iupbs01(msgin,'HOUR'), 8, msgot, ibit )
106  CALL pkb ( iupbs01(msgin,'MINU'), 8, msgot, ibit )
107 
108 C Set a default of 0 for the second.
109 
110  CALL pkb ( 0, 8, msgot, ibit )
111 
112 C Copy Section 2 (if it exists) through the next-to-last byte
113 C of Section 3 from the input array to the output array.
114 
115  CALL mvb ( msgin, iad2+1, msgot, (ibit/8)+1, len2+len3-1 )
116 
117 C Store the length of the new Section 3.
118 
119  ibit = ( len0 + len1ot + len2 ) * 8
120  CALL pkb ( len3ot, 24, msgot, ibit )
121 
122 C Copy Section 4 and Section 5 from the input array to the
123 C output array.
124 
125  ibit = ibit + ( len3ot * 8 ) - 24
126  CALL mvb ( msgin, iad4+1, msgot, (ibit/8)+1, lenm-iad4 )
127 
128  RETURN
