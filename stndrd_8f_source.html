4 C> THIS SUBROUTINE READS AN INPUT NCEP BUFR MESSAGE CONTAINED
5 C> WITHIN ARRAY MSGIN AND, USING THE BUFR TABLES INFORMATION ASSOCIATED
6 C> WITH LOGICAL UNIT LUNIT, OUTPUTS A "STANDARDIZED" VERSION OF THIS
7 C> SAME MESSAGE WITHIN ARRAY MSGOT. THIS "STANDARDIZATION" INVOLVES
8 C> REMOVING ALL OCCURRENCES OF NCEP BUFRLIB-SPECIFIC BYTE COUNTERS AND
9 C> BIT PADS IN SECTION 4 AS WELL AS REPLACING THE TOP-LEVEL TABLE A FXY
10 C> NUMBER IN SECTION 3 WITH AN EQUIVALENT SEQUENCE OF LOWER-LEVEL
11 C> TABLE B, TABLE C, TABLE D AND/OR REPLICATION FXY NUMBERS WHICH
12 C> DIRECTLY CONSTITUTE THAT TABLE A FXY NUMBER AND WHICH THEMSELVES ARE
13 C> ALL WMO-STANDARD. THE RESULT IS THAT THE OUTPUT MESSAGE IN MSGOT IS
14 C> NOW ENTIRELY COMPLIANT WITH WMO FM-94 BUFR REGULATIONS (I.E. IT IS
15 C> NOW "STANDARD"). IT IS IMPORTANT TO NOTE THAT THE SEQUENCE EXPANSION
16 C> WITHIN SECTION 3 MAY CAUSE THE FINAL "STANDARDIZED" BUFR MESSAGE TO
17 C> BE LONGER THAN THE ORIGINAL INPUT NCEP BUFR MESSAGE BY AS MANY AS
18 C> (MAXNC*2) BYTES (SEE 'bufrlib.prm' FOR AN EXPLANATION OF MAXNC), SO
19 C> THE USER MUST ALLOW FOR ENOUGH SPACE TO ACCOMODATE SUCH AN EXPANSION
20 C> WITHIN THE MSGOT ARRAY.
21 C>
22 C> PROGRAM HISTORY LOG:
23 C> 2004-08-18 J. ATOR -- ORIGINAL AUTHOR
24 C> THIS SUBROUTINE IS MODELED AFTER SUBROUTINE
25 C> STANDARD; HOWEVER, IT USES SUBROUTINE RESTD
26 C> TO EXPAND SECTION 3 AS MANY LEVELS AS
27 C> NECESSARY IN ORDER TO ATTAIN TRUE WMO
28 C> STANDARDIZATION (WHEREAS STANDARD ONLY
29 C> EXPANDED THE TOP-LEVEL TABLE A FXY NUMBER
30 C> ONE LEVEL DEEP), AND IT ALSO CONTAINS AN
31 C> EXTRA INPUT ARGUMENT LMSGOT WHICH PREVENTS
32 C> OVERFLOW OF THE MSGOT ARRAY
33 C> 2005-11-29 J. ATOR -- USE GETLENS AND IUPBS01; ENSURE THAT BYTE 4
34 C> OF SECTION 4 IS ZEROED OUT IN MSGOT; CHECK
35 C> EDITION NUMBER OF BUFR MESSAGE BEFORE
36 C> PADDING TO AN EVEN BYTE COUNT
37 C> 2009-03-23 J. ATOR -- USE IUPBS3 AND NEMTBAX; DON'T ASSUME THAT
38 C> COMPRESSED MESSAGES ARE ALREADY FULLY
39 C> STANDARDIZED WITHIN SECTION 3
40 C> 2014-02-04 J. ATOR -- ACCOUNT FOR SUBSETS WITH BYTE COUNT > 65530
41 C>
42 C> USAGE: CALL STNDRD (LUNIT, MSGIN, LMSGOT, MSGOT)
43 C> INPUT ARGUMENT LIST:
44 C> LUNIT - INTEGER: FORTRAN LOGICAL UNIT NUMBER FOR BUFR FILE
45 C> MSGIN - INTEGER: *-WORD ARRAY CONTAINING BUFR MESSAGE IN NCEP
46 C> BUFR
47 C> LMSGOT - INTEGER: DIMENSIONED SIZE (IN INTEGER WORDS) OF MSGOT;
48 C> USED BY THE SUBROUTINE TO ENSURE THAT IT DOES NOT
49 C> OVERFLOW THE MSGOT ARRAY
50 C>
51 C> OUTPUT ARGUMENT LIST:
52 C> MSGOT - INTEGER: *-WORD ARRAY CONTAINING INPUT BUFR MESSAGE
53 C> NOW IN STANDARDIZED BUFR
54 C>
55 C> REMARKS:
56 C> MSGIN AND MSGOT MUST BE SEPARATE ARRAYS.
57 C>
58 C> THIS ROUTINE CALLS: BORT GETLENS ISTDESC IUPB
59 C> IUPBS01 IUPBS3 MVB NEMTBAX
60 C> NUMTAB PKB PKC RESTD

62 C> THIS ROUTINE IS CALLED BY: MSGWRT
63 C> Also called by application programs.
64 C>
65  SUBROUTINE stndrd(LUNIT,MSGIN,LMSGOT,MSGOT)

69  include 'bufrlib.prm'
70 
71  dimension icd(maxnc)
72 
73  COMMON /hrdwrd/ nbytw,nbitw,iord(8)
74 
75  dimension msgin(*),msgot(*)
76 
77  CHARACTER*128 BORT_STR
78  CHARACTER*8 SUBSET
79  CHARACTER*4 SEVN
80  CHARACTER*1 TAB
81 
82  LOGICAL FOUND
83 
84 C-----------------------------------------------------------------------
85 C-----------------------------------------------------------------------
86 
87 C LUNIT MUST POINT TO AN OPEN BUFR FILE
88 C -------------------------------------
89 
90  CALL status(lunit,lun,il,im)
91  IF(il.EQ.0) GOTO 900
92 
93 C IDENTIFY THE SECTION LENGTHS AND ADDRESSES IN MSGIN
94 C ---------------------------------------------------
95 
96  CALL getlens(msgin,5,len0,len1,len2,len3,len4,len5)
97 
98  iad3 = len0+len1+len2
99  iad4 = iad3+len3
100 
101  lenn = len0+len1+len2+len3+len4+len5
102 
103  lenm = iupbs01(msgin,'LENM')
104 
105  IF(lenn.NE.lenm) GOTO 901
106 
107  mbit = (lenn-4)*8
108  CALL upc(sevn,4,msgin,mbit,.true.)
109  IF(sevn.NE.'7777') GOTO 902
110 
111 C COPY SECTIONS 0 THROUGH PART OF SECTION 3 INTO MSGOT
112 C ----------------------------------------------------
113 
114  mxbyto = (lmsgot*nbytw) - 8
115 
116  lbyto = iad3+7
117  IF(lbyto.GT.mxbyto) GOTO 905
118  CALL mvb(msgin,1,msgot,1,lbyto)
119 
120 C REWRITE NEW SECTION 3 IN A "STANDARD" FORM
121 C ------------------------------------------
122 
123 C LOCATE THE TOP-LEVEL TABLE A DESCRIPTOR
124 
125  found = .false.
126  ii = 10
127  DO WHILE ((.NOT.found).AND.(ii.GE.8))
128  isub = iupb(msgin,iad3+ii,16)
129  CALL numtab(lun,isub,subset,tab,itab)
130  IF((itab.NE.0).AND.(tab.EQ.'D')) THEN
131  CALL nemtbax(lun,subset,mtyp,msbt,inod)
132  IF(inod.NE.0) found = .true.
133  ENDIF
134  ii = ii - 2
135  ENDDO
136  IF(.NOT.found) GOTO 903
137 
138  IF (istdesc(isub).EQ.0) THEN
139 
140 C ISUB IS A NON-STANDARD TABLE A DESCRIPTOR AND NEEDS
141 C TO BE EXPANDED INTO AN EQUIVALENT STANDARD SEQUENCE
142 
143  CALL restd(lun,isub,ncd,icd)
144  ELSE
145 
146 C ISUB IS ALREADY A STANDARD DESCRIPTOR, SO JUST COPY
147 C IT "AS IS" INTO THE NEW SECTION 3 (I.E. NO EXPANSION
148 C IS NECESSARY!)
149 
150  ncd = 1
151  icd(ncd) = isub
152  ENDIF
153 
154 C USE THE EDITION NUMBER TO DETERMINE THE LENGTH OF THE
155 C NEW SECTION 3
156 
157  len3 = 7+(ncd*2)
158  iben = iupbs01(msgin,'BEN')
159  IF(iben.LT.4) THEN
160  len3 = len3+1
161  ENDIF
162  lbyto = lbyto + len3 - 7
163  IF(lbyto.GT.mxbyto) GOTO 905
164 
165 C STORE THE DESCRIPTORS INTO THE NEW SECTION 3
166 
167  ibit = (iad3+7)*8
168  DO n=1,ncd
169  CALL pkb(icd(n),16,msgot,ibit)
170  ENDDO
171 
172 C DEPENDING ON THE EDITION NUMBER, PAD OUT THE NEW SECTION 3 WITH AN
173 C ADDITIONAL ZEROED-OUT BYTE IN ORDER TO ENSURE AN EVEN BYTE COUNT
174 
175  IF(iben.LT.4) THEN
176  CALL pkb(0,8,msgot,ibit)
177  ENDIF
178 
179 C STORE THE LENGTH OF THE NEW SECTION 3
180 
181  ibit = iad3*8
182  CALL pkb(len3,24,msgot,ibit)
183 
184 C NOW THE TRICKY PART - NEW SECTION 4
185 C -----------------------------------
186 
187  IF(iupbs3(msgin,'ICMP').EQ.1) THEN
188 
189 C THE DATA IN SECTION 4 IS COMPRESSED AND IS THEREFORE ALREADY
190 C STANDARDIZED, SO COPY IT "AS IS" INTO THE NEW SECTION 4
191 
192  IF((lbyto+len4+4).GT.mxbyto) GOTO 905
193 
194  CALL mvb(msgin,iad4+1,msgot,lbyto+1,len4)
195 
196  jbit = (lbyto+len4)*8
197 
198  ELSE
199 
200  nad4 = iad3+len3
201 
202  ibit = (iad4+4)*8
203  jbit = (nad4+4)*8
204 
205  lbyto = lbyto + 4
206 
207 C COPY THE SUBSETS, MINUS THE BYTE COUNTERS AND BIT PADS, INTO
208 C THE NEW SECTION 4
209 
210  nsub = iupbs3(msgin,'NSUB')
211 
212  DO 10 i=1,nsub
213  CALL upb(lsub,16,msgin,ibit)
214  IF(nsub.GT.1) THEN
215 
216 C USE THE BYTE COUNTER TO COPY THIS SUBSET
217 
218  islen = lsub-2
219  ELSE
220 
221 C THIS IS THE ONLY SUBSET IN THE MESSAGE, AND IT COULD
222 C POSSIBLY BE AN OVERLARGE (> 65530 BYTES) SUBSET, IN
223 C WHICH CASE WE CAN'T RELY ON THE VALUE STORED IN THE
224 C BYTE COUNTER. EITHER WAY, WE DON'T REALLY NEED IT.
225 
226  islen = iad4+len4-(ibit/8)
227  ENDIF
228  DO l=1,islen
229  CALL upb(nval,8,msgin,ibit)
230  lbyto = lbyto + 1
231  IF(lbyto.GT.mxbyto) GOTO 905
232  CALL pkb(nval,8,msgot,jbit)
233  ENDDO
234  DO k=1,8
235  kbit = ibit-k-8
236  CALL upb(kval,8,msgin,kbit)
237  IF(kval.EQ.k) THEN
238  jbit = jbit-k-8
239  GOTO 10
240  ENDIF
241  ENDDO
242  GOTO 904
243 10 ENDDO
244 
245 C FROM THIS POINT ON, WE WILL NEED (AT MOST) 6 MORE BYTES OF
246 C SPACE WITHIN MSGOT IN ORDER TO BE ABLE TO STORE THE ENTIRE
247 C STANDARDIZED MESSAGE (I.E. WE WILL NEED (AT MOST) 2 MORE
248 C ZEROED-OUT BYTES IN SECTION 4 PLUS THE 4 BYTES '7777' IN
249 C SECTION 5), SO DO A FINAL MSGOT OVERFLOW CHECK NOW.
250 
251  IF(lbyto+6.GT.mxbyto) GOTO 905
252 
253 C PAD THE NEW SECTION 4 WITH ZEROES UP TO THE NEXT WHOLE BYTE
254 C BOUNDARY.
255 
256  DO WHILE(.NOT.(mod(jbit,8).EQ.0))
257  CALL pkb(0,1,msgot,jbit)
258  ENDDO
259 
260 C DEPENDING ON THE EDITION NUMBER, WE MAY NEED TO FURTHER PAD
261 C THE NEW SECTION 4 WITH AN ADDITIONAL ZEROED-OUT BYTE IN ORDER
262 C TO ENSURE THAT THE PADDING IS UP TO AN EVEN BYTE BOUNDARY.
263 
264  IF( (iben.LT.4) .AND. (mod(jbit/8,2).NE.0) ) THEN
265  CALL pkb(0,8,msgot,jbit)
266  ENDIF
267 
268  ibit = nad4*8
269  len4 = jbit/8 - nad4
270  CALL pkb(len4,24,msgot,ibit)
271  CALL pkb(0,8,msgot,ibit)
272  ENDIF
273 
274 C FINISH THE NEW MESSAGE WITH AN UPDATED SECTION 0 BYTE COUNT
275 C -----------------------------------------------------------
276 
277  ibit = 32
278  lenn = len0+len1+len2+len3+len4+len5
279  CALL pkb(lenn,24,msgot,ibit)
280 
281  CALL pkc('7777',4,msgot,jbit)
282 
283 C EXITS
284 C -----
285 
286  RETURN
287 900 CALL bort('BUFRLIB: STNDRD - BUFR FILE IS CLOSED, IT MUST BE'//
288  . ' OPEN')
289 901 WRITE(bort_str,'("BUFRLIB: STNDRD - INPUT MESSAGE LENGTH FROM'//
290  . ' SECTION 0",I6," DOES NOT EQUAL SUM OF ALL INDIVIDUAL SECTION'//
291  . ' LENGTHS (",I6,")")') lenm,lenn
292  CALL bort(bort_str)
293 902 WRITE(bort_str,'("BUFRLIB: STNDRD - INPUT MESSAGE DOES NOT '//
294  . 'END WITH ""7777"" (ENDS WITH ",A)') sevn
295  CALL bort(bort_str)
296 903 CALL bort('BUFRLIB: STNDRD - TABLE A SUBSET DESCRIPTOR '//
297  . 'NOT FOUND')
298 904 CALL bort('BUFRLIB: STNDRD - BIT MISMATCH COPYING SECTION 4 '//
299  . 'FROM INPUT TO OUTPUT (STANDARD) MESSAGE')
300 905 CALL bort('BUFRLIB: STNDRD - OVERFLOW OF OUTPUT (STANDARD) '//
301  . 'MESSAGE ARRAY; TRY A LARGER DIMENSION FOR THIS ARRAY')
302  END
