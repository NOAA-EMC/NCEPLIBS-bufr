4 C> THIS SUBROUTINE PERFORMS SOME FINAL CHECKS ON AN OUTPUT
5 C> BUFR MESSAGE (E.G., CONFIRMING THAT EACH SECTION OF THE MESSAGE HAS
6 C> AN EVEN NUMBER OF BYTES WHEN NECESSARY, "STANDARDIZING" THE MESSAGE
7 C> IF REQUESTED VIA A PREVIOUS CALL TO BUFR ARCHIVE LIBRARY SUBROUTINE
8 C> STDMSG, ETC.), AND THEN PREPARES THE MESSAGE FOR FINAL OUTPUT TO
9 C> LOGICAL UNIT LUNIT (E.G., ADDING THE STRING "7777" TO THE LAST FOUR
10 C> BYTES OF THE MESSAGE, APPENDING ZEROED-OUT BYTES UP TO A SUBSEQUENT
11 C> MACHINE WORD BOUNDARY, ETC.). IT THEN WRITES OUT THE FINISHED
12 C> MESSAGE TO LOGICAL UNIT LUNIT AND ALSO STORES A COPY OF IT WITHIN
13 C> MODULE BUFRMG FOR POSSIBLE LATER RETRIEVAL VIA BUFR ARCHIVE
14 C> LIBRARY SUBROUTINE WRITSA.
15 C>

92  COMMON /msgstd/ csmf
93  COMMON /tnkrcp/ itryr,itrmo,itrdy,itrhr,itrmi,ctrt

95  CHARACTER*128 ERRSTR
96 
97  CHARACTER*4 BUFR,SEVN
98  CHARACTER*1 CSMF
99  CHARACTER*1 CTRT
100  dimension mesg(*)
101  dimension iec0(2)
102 
103  DATA bufr/'BUFR'/
104  DATA sevn/'7777'/
105 
106 C-----------------------------------------------------------------------
107 C-----------------------------------------------------------------------
108 
109 C MAKE A LOCAL COPY OF THE INPUT MESSAGE FOR USE WITHIN THIS
110 C SUBROUTINE, SINCE CALLS TO ANY OR ALL OF THE SUBROUTINES STNDRD,
111 C CNVED4, PKBS1, ATRCPT, ETC. MAY END UP MODIFYING THE MESSAGE
112 C BEFORE IT FINALLY GETS WRITTEN OUT TO LUNIT.
113 
114  mbyt = mgbyt
115 
116  iec0(1) = mesg(1)
117  iec0(2) = mesg(2)
118  ibit = 32
119  CALL pkb(mbyt,24,iec0,ibit)
120 
121  DO ii = 1, nmwrd(iec0)
122  mgwa(ii) = mesg(ii)
123  ENDDO
124 
125 C OVERWRITE ANY VALUES WITHIN SECTION 0 OR SECTION 1 THAT WERE
126 C REQUESTED VIA PREVIOUS CALLS TO BUFR ARCHIVE LIBRARY SUBROUTINE
127 C PKVS01. IF A REQUEST WAS MADE TO CHANGE THE BUFR EDITION NUMBER
128 C TO 4, THEN ACTUALLY CONVERT THE MESSAGE AS WELL.
129 
130  IF(ns01v.GT.0) THEN
131  DO i=1,ns01v
132  IF(cmnem(i).EQ.'BEN') THEN
133  IF(ivmnem(i).EQ.4) THEN
134 
135 C INSTALL SECTION 0 BYTE COUNT FOR USE BY SUBROUTINE CNVED4.
136 
137  ibit = 32
138  CALL pkb(mbyt,24,mgwa,ibit)
139 
140  CALL cnved4(mgwa,mxmsgld4,mgwb)
141 
142 C COMPUTE MBYT FOR THE NEW EDITION 4 MESSAGE.
143 
144  mbyt = iupbs01(mgwb,'LENM')
145 
146 C COPY THE MGWB ARRAY BACK INTO MGWA.
147 
148  DO ii = 1, nmwrd(mgwb)
149  mgwa(ii) = mgwb(ii)
150  ENDDO
151  ENDIF
152  ELSE
153 
154 C OVERWRITE THE REQUESTED VALUE.
155 
156  CALL pkbs1(ivmnem(i),mgwa,cmnem(i))
157  ENDIF
158  ENDDO
159  ENDIF
160 
161 C "STANDARDIZE" THE MESSAGE IF REQUESTED VIA COMMON /MSGSTD/.
162 C HOWEVER, WE DO NOT WANT TO DO THIS IF THE MESSAGE CONTAINS BUFR
163 C TABLE (DX) INFORMATION, IN WHICH CASE IT IS ALREADY "STANDARD".
164 
165  IF ( ( csmf.EQ.'Y' ) .AND. ( idxmsg(mgwa).NE.1 ) ) THEN
166 
167 C INSTALL SECTION 0 BYTE COUNT AND SECTION 5 '7777' INTO THE
168 C ORIGINAL MESSAGE. THIS IS NECESSARY BECAUSE SUBROUTINE STNDRD
169 C REQUIRES A COMPLETE AND WELL-FORMED BUFR MESSAGE AS ITS INPUT.
170 
171  ibit = 32
172  CALL pkb(mbyt,24,mgwa,ibit)
173  ibit = (mbyt-4)*8
174  CALL pkc(sevn,4,mgwa,ibit)
175 
176  CALL stndrd(lunit,mgwa,mxmsgld4,mgwb)
177 
178 C COMPUTE MBYT FOR THE NEW "STANDARDIZED" MESSAGE.
179 
180  mbyt = iupbs01(mgwb,'LENM')
181 
182 C COPY THE MGWB ARRAY BACK INTO MGWA.
183 
184  DO ii = 1, nmwrd(mgwb)
185  mgwa(ii) = mgwb(ii)
186  ENDDO
187  ENDIF
188 
189 C APPEND THE TANK RECEIPT TIME TO SECTION 1 IF REQUESTED VIA
190 C COMMON /TNKRCP/, UNLESS THE MESSAGE CONTAINS BUFR TABLE (DX)
191 C INFORMATION.
192 
193  IF ( ( ctrt.EQ.'Y' ) .AND. ( idxmsg(mgwa).NE.1 ) ) THEN
194 
195 C INSTALL SECTION 0 BYTE COUNT FOR USE BY SUBROUTINE ATRCPT.
196 
197  ibit = 32
198  CALL pkb(mbyt,24,mgwa,ibit)
199 
200  CALL atrcpt(mgwa,mxmsgld4,mgwb)
201 
202 C COMPUTE MBYT FOR THE REVISED MESSAGE.
203 
204  mbyt = iupbs01(mgwb,'LENM')
205 
206 C COPY THE MGWB ARRAY BACK INTO MGWA.
207 
208  DO ii = 1, nmwrd(mgwb)
209  mgwa(ii) = mgwb(ii)
210  ENDDO
211  ENDIF
212 
213 C GET THE SECTION LENGTHS.
214 
215  CALL getlens(mgwa,4,len0,len1,len2,len3,len4,l5)
216 
217 C DEPENDING ON THE EDITION NUMBER OF THE MESSAGE, WE NEED TO ENSURE
218 C THAT EACH SECTION WITHIN THE MESSAGE HAS AN EVEN NUMBER OF BYTES.
219 
220  IF(iupbs01(mgwa,'BEN').LT.4) THEN
221  IF(mod(len1,2).NE.0) GOTO 901
222  IF(mod(len2,2).NE.0) GOTO 902
223  IF(mod(len3,2).NE.0) GOTO 903
224  IF(mod(len4,2).NE.0) THEN
225 
226 C PAD SECTION 4 WITH AN ADDITIONAL BYTE
227 C THAT IS ZEROED OUT.
228 
229  iad4 = len0+len1+len2+len3
230  iad5 = iad4+len4
231  ibit = iad4*8
232  len4 = len4+1
233  CALL pkb(len4,24,mgwa,ibit)
234  ibit = iad5*8
235  CALL pkb(0,8,mgwa,ibit)
236  mbyt = mbyt+1
237  ENDIF
238  ENDIF
239 
240 C WRITE SECTION 0 BYTE COUNT AND SECTION 5
241 C ----------------------------------------
242 
243  ibit = 0
244  CALL pkc(bufr, 4,mgwa,ibit)
245  CALL pkb(mbyt,24,mgwa,ibit)
246 
247  kbit = (mbyt-4)*8
248  CALL pkc(sevn, 4,mgwa,kbit)
249 
250 C ZERO OUT THE EXTRA BYTES WHICH WILL BE WRITTEN
251 C ----------------------------------------------
252 
253 C I.E. SINCE THE BUFR MESSAGE IS STORED WITHIN THE INTEGER ARRAY
254 C MGWA(*) (RATHER THAN WITHIN A CHARACTER ARRAY), WE NEED TO MAKE
255 C SURE THAT THE "7777" IS FOLLOWED BY ZEROED-OUT BYTES UP TO THE
256 C BOUNDARY OF THE LAST MACHINE WORD THAT WILL BE WRITTEN OUT.
257 
258  CALL padmsg(mgwa,mxmsgld4,npbyt)
259 
260 C WRITE THE MESSAGE PLUS PADDING TO A WORD BOUNDARY IF NULL(LUN) = 0
261 C ------------------------------------------------------------------
262 
263  mwrd = nmwrd(mgwa)
264  CALL status(lunit,lun,il,im)
265  IF(null(lun).EQ.0) THEN
266  CALL blocks(mgwa,mwrd)
267  CALL cwrbufr(lun,mgwa,mwrd)
268  ENDIF
269 
270  IF(iprt.GE.2) THEN
271  CALL errwrt('++++++++++++++BUFR ARCHIVE LIBRARY+++++++++++++++++')
272  WRITE ( unit=errstr, fmt='(A,I4,A,I7)')
273  . 'BUFRLIB: MSGWRT: LUNIT =', lunit, ', BYTES =', mbyt+npbyt
274  CALL errwrt(errstr)
275  CALL errwrt('++++++++++++++BUFR ARCHIVE LIBRARY+++++++++++++++++')
276  CALL errwrt(' ')
277  ENDIF
278 
279 C SAVE A MEMORY COPY OF THIS MESSAGE, UNLESS IT'S A DX MESSAGE
280 C ------------------------------------------------------------
281 
282  IF(idxmsg(mgwa).NE.1) THEN
283 
284 C STORE A COPY OF THIS MESSAGE WITHIN MODULE BUFRMG,
285 C FOR POSSIBLE LATER RETRIEVAL DURING A FUTURE CALL TO
286 C SUBROUTINE WRITSA.
287 
288  msglen(lun) = mwrd
289  DO i=1,msglen(lun)
290  msgtxt(i,lun) = mgwa(i)
291  ENDDO
292  ENDIF
293 
294 C EXITS
295 C -----
296 
297  RETURN

299  . ('BUFRLIB: MSGWRT - LENGTH OF SECTION 1 IS NOT A MULTIPLE OF 2')
300 902 CALL bort
301  . ('BUFRLIB: MSGWRT - LENGTH OF SECTION 2 IS NOT A MULTIPLE OF 2')
302 903 CALL bort
303  . ('BUFRLIB: MSGWRT - LENGTH OF SECTION 3 IS NOT A MULTIPLE OF 2')
304  END
