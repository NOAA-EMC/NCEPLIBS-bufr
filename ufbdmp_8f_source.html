4 C> THIS SUBROUTINE DUMPS A DETAILED PRINT LISTING OF THE
5 C> CONTENTS OF THE UNPACKED DATA SUBSET CURRENTLY RESIDING IN THE
6 C> INTERNAL ARRAYS ASSOCIATED WITH THE BUFR FILE IN LOGICAL UNIT
7 C> ABS(LUNIN). ABS(LUNIN) MUST HAVE BEEN OPENED FOR INPUT VIA A
8 C> PREVIOUS CALL TO BUFR ARCHIVE LIBRARY SUBROUTINE OPENBF. THE DATA
9 C> SUBSET MUST HAVE BEEN SUBSEQUENTLY READ INTO THE INTERNAL BUFR
10 C> ARCHIVE LIBRARY ARRAYS VIA A CALL TO BUFR ARCHIVE LIBRARY SUBROUTINE
11 C> READMG OR READERME, FOLLOWED BY A CALL TO BUFR ARCHIVE LIBRARY
12 C> SUBROUTINE READSB (OR VIA A SINGLE CALL TO BUFR ARCHIVE LIBRARY
13 C> SUBROUTINE READNS!). FOR A PARTICULAR SUBSET, THE PRINT LISTING
14 C> CONTAINS EACH MNEMONIC ACCOMPANIED BY ITS CORRESPONDING DATA VALUE
15 C> (INCLUDING THE ACTUAL BITS THAT WERE SET FOR FLAG TABLE VALUES!)
16 C> ALONG WITH OTHER POTENTIALLY USEFUL INFORMATION SUCH AS WHICH OTHER
17 C> MNEMONIC(S) THAT MNEMONIC WAS A CONSTITUENT OF WITHIN THE OVERALL
18 C> DATA SUBSET. HOWEVER, THE LISTING ALSO CONTAINS OTHER MORE ESOTERIC
19 C> INFORMATION SUCH AS BUFR STORAGE CHARACTERISTICS AND A COPY OF THE
20 C> JUMP/LINK TABLE USED INTERNALLY WITHIN THE BUFR ARCHIVE LIBRARY
21 C> SOFTWARE. THIS SUBROUTINE IS SIMILAR TO BUFR ARCHIVE LIBRARY
22 C> SUBROUTINE UFDUMP, EXCEPT THAT UFDUMP DOES NOT PRINT POINTERS,
23 C> COUNTERS AND THE OTHER MORE ESOTERIC INFORMATION DESCRIBING THE
24 C> INTERNAL SUBSET STRUCTURES. EACH SUBROUTINE, UFBDMP AND UFDUMP,
25 C> IS USEFUL FOR DIFFERENT DIAGNOSTIC PURPOSES, BUT IN GENERAL UFDUMP
26 C> IS MORE USEFUL FOR JUST LOOKING AT THE DATA ELEMENTS.
27 C>

96 C> REMARKS:

108 C> THIS SUBROUTINE STOPS THE SCROLL AND RETURNS TO THE CALLING
109 C> PROGRAM (PRESUMABLY TO READ IN THE NEXT SUBSET IN THE BUFR FILE).

198  rj = rjust(tg_rj)

202  CALL nemtab(lun,tg,idn,tab,n)
203  IF(tabb(n,lun)(71:75).EQ.'FLAG') THEN
204 
205 C Print a listing of the bits corresponding to
206 C this value.
207 
208  CALL upftbv(lunit,tg,vl,mxfv,ifv,nifv)
209  IF(nifv.GT.0) THEN
210  bits(1:1) = '('
211  ipt = 2
212  DO ii=1,nifv
213  isz = isize(ifv(ii))
214  WRITE(fmtf,'(A2,I1,A4)') '(I', isz, ',A1)'
215  IF((ipt+isz).LE.14) THEN
216  WRITE(bits(ipt:ipt+isz),fmtf) ifv(ii), ','
217  ipt = ipt + isz + 1
218  ELSE
219  bits(2:13) = 'MANY BITS ON'
220  ipt = 15
221  ENDIF
222  ENDDO
223  bits(ipt-1:ipt-1) = ')'
224  ENDIF
225  ENDIF
226  ENDIF
227  IF(ibfms(vl).NE.0) THEN
228  lchr = 'MISSING'
229  rj = rjust(lchr)
230  WRITE(luout,2) nv,tp,it,tg_rj,lchr,ib,is,ir,nd,jp,lk,jb
231  ELSE
232  IF(lunit.EQ.lunin) THEN
233  WRITE(luout,1) nv,tp,it,tg_rj,vl,bits,ib,is,ir,nd,jp,lk,
234  . jb
235  ELSE
236  WRITE(luout,10) nv,tp,it,tg_rj,vl,bits,ib,is,ir,nd,jp,lk,
237  . jb
238  ENDIF
239  ENDIF
240  ELSE
241  IF(ib.GT.64) THEN
242  CALL readlc(lunit,lchr,tg_rj)
243  ELSE
244  lchr = vc
245  ENDIF
246  IF(ibfms(vl).NE.0) lchr = 'MISSING'
247  rj = rjust(lchr)
248  WRITE(luout,2) nv,tp,it,tg_rj,lchr,ib,is,ir,nd,jp,lk,jb
249  ENDIF
250  ENDDO
251 
252  WRITE(luout,3)
253 
254 1 FORMAT(i5,1x,a3,'-',i1,1x,a10,5x,g15.6,1x,a14,7(1x,i5))
255 10 FORMAT(i5,1x,a3,'-',i1,1x,a10,5x,f15.6,1x,a14,7(1x,i5))
256 2 FORMAT(i5,1x,a3,'-',i1,1x,a10, a20, 15x, 7(1x,i5))
257 3 FORMAT(/' >>> END OF SUBSET <<< '/)
258

261
