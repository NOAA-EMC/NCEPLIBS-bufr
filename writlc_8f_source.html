123

128  ENDIF
129  ENDIF
130  ENDDO
131  ELSE
132

135

139 
140 C Find the most recently written subset in the message.
141

143  ibit = mbyte*8
144  CALL upb(nbyt,16,mbay(1,lun),ibit)

147  ENDDO
148 
149  IF(nsubs.NE.nsub(lun)) THEN

157  ENDIF

159  ENDIF
160

162 
163  itagct = 0
164  mbit = mbyte*8 + 16
165  nbit = 0
166  n = 1
167  CALL usrtpl(lun,n,n)
168  DO WHILE (n+1.LE.nval(lun))
169  n = n+1
170  node = inv(n,lun)
171  mbit = mbit+nbit
172  nbit = ibt(node)
173  IF(itp(node).EQ.1) THEN
174  CALL upbb(ival,nbit,mbit,mbay(1,lun))
175  CALL usrtpl(lun,n,ival)
176  ELSEIF(ctag.EQ.tag(node)) THEN
177  itagct = itagct + 1
178  IF(itagct.EQ.ioid) THEN
179  IF(itp(node).NE.3) goto 904
180  nchr = nbit/8
181  ibit = mbit
182  DO j=1,nchr
183  CALL pkc(' ',1,mbay(1,lun),ibit)
184  ENDDO
185  CALL pkc(chr,nchr,mbay(1,lun),mbit)
186  CALL usrtpl(lun,1,1)
187  goto 100
188  ENDIF
189  ENDIF
190  ENDDO
191  ENDIF
192 
193  IF(iprt.GE.0) THEN
194  CALL errwrt('+++++++++++++++++++++WARNING+++++++++++++++++++++++')
195  errstr = 'BUFRLIB: WRITLC - COULDN''T WRITE VALUE FOR ' // ctag
196  . // ' INTO SUBSET, BECAUSE IT WASN''T FOUND IN THE SUBSET'
197  . // ' DEFINITION'
198  CALL errwrt(errstr)
199  errstr = '(' // ctag // ' MAY NOT BE IN THE BUFR TABLE(?))'
200  CALL errwrt(errstr)
201  CALL errwrt('+++++++++++++++++++++WARNING+++++++++++++++++++++++')
202  CALL errwrt(' ')
203  ENDIF
204
